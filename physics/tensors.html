\begin{align} U &= U_1\hat{e}_1 + U_2\hat{e}_2 + \cdots + U_n\hat{e}_n = \sum_i^n{U_i\hat{e}_i}\\ V &= V_1\hat{e}_1 + V_2\hat{e}_2 + \cdots + V_n\hat{e}_n = \sum_i^n{V_i\hat{e}_i}\\ UV &= (U_1\hat{e}_1 + U_2\hat{e}_2 + \cdots + U_n\hat{e}_n)(V_1\hat{e}_1 + V_2\hat{e}_2 + \cdots + V_n\hat{e}_n)\\ &= \begin{array}{c} U_1V_1\hat{e}_1\hat{e}_1 &+& U_1V_2\hat{e}_1\hat{e}_2 &+& \cdots &+& U_1V_n\hat{e}_1\hat{e}_n &+ \\ U_2V_1\hat{e}_2\hat{e}_1 &+& U_2V_2\hat{e}_2\hat{e}_2 &+& \cdots &+& U_1V_n\hat{e}_1\hat{e}_n &+ \\ \vdots &+& \vdots &+& \ddots &+& \vdots &+ \\ U_nV_1\hat{e}_n\hat{e}_1 &+& U_nV_2\hat{e}_n\hat{e}_2 &+& \cdots &+& U_nV_n\hat{e}_n\hat{e}_n \end{array}\\ &= \left(\sum_i^n{U_i\hat{e}_i}\right)\left(\sum_i^n{V_i\hat{e}_i}\right) = \sum_i^n{U_i\hat{e}_i}\sum_j^n{V_j\hat{e}_j} = \sum_i^n\sum_j^n{U_iV_j\hat{e}_i\hat{e}_j}\\ \mu_{ij} &= U_iV_j\hat{e}_i\hat{e}_j \end{align}

Covariant and Contravariant vectors

\begin{align} i \cdot i^* &= 1\\ i \cdot j^* &= 0\\ j \cdot i^* &= 0\\ j \cdot j^* &= 1\\ i &= \langle a, b \rangle\\ i* &= \langle c, d \rangle\\ j &= \langle e, f \rangle\\ j* &= \langle g, h \rangle\\ ac + bd &= 1\\ ag + bh &= 0\\ ec + fd &= 0\\ eg + fh &= 1\\ a &= {{1 - bd} \over {c}}\\ c &= {g(bd - 1) \over bh}\\ fh &= {eg(1 - bd) \over db}\\ eg &= db\\ a &= {5 \over {c}}\\ c &= {5g \over bh}\\ fh &= 5\\ 2\cdot3 &= 6\cdot1\\ ag &= bh\\ 3a &= 1h\\ \left(\begin{array}{cc} a & b\\ e & f\\ \end{array}\right) \left(\begin{array}{cc} c & g\\ d & h\\ \end{array}\right) &= \left(\begin{array}{cc} 1 & 0\\ 0 & 1\\ \end{array}\right)\\ \left(\begin{array}{cc} 3 & 1\\ 2 & 5\\ \end{array}\right) \left(\begin{array}{cc} 15 & 3\\ 6 & 1\\ \end{array}\right) &= \left(\begin{array}{cc} 1 & 0\\ 0 & 1\\ \end{array}\right) \end{align}
