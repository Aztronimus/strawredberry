Energy

Mass-energy conversion equation

\begin{align} K &= m G(v)\\\\ E_2 &= 2hf + K_2\\ E_1 &= K_1 + hf\left(\sqrt{\frac{1 - V}{1 + V}} + \sqrt{\frac{1 + V}{1 - V}}\right)\\\\ 2hf + K_2 &= K_1 + hf\left(\frac{1 - V}{\sqrt{1 - V^2}} + \frac{1 + V}{\sqrt{1 - V^2}}\right)\\ 2hf + K_2 &= K_1 + \frac{2hf}{\sqrt{1 - V^2}}\\ K_2 - K_1 &= 2hf\left(\frac{1}{\sqrt{1 - V^2}} - 1\right)\\ (m_2 - m_1)G &= 2hf\left(\frac{1}{\sqrt{1 - V^2}} - 1\right)\\\\ \frac{E}{m} &= \frac{G}{\frac{1}{\sqrt{1 - V^2}} - 1}\\ G &= k\left(\frac{1}{\sqrt{1 - V^2}} - 1\right)\\ \\ K &= mk\left(\frac{1}{\sqrt{1 - V^2}} - 1\right)\\ E &= mk \end{align}

Momentum of a photon

\begin{align} \frac{P_{Bx}}{P_B} &= \frac{v}{c} = V\\ \frac{P_{By}}{P_B} &= \frac{\sqrt{c^2 - v^2}}{c} = \sqrt{1 - \frac{v^2}{c^2}} = \frac{1}{\gamma}\\ \\ P &= P(E)\\ \\ E_A &= hf\\ E_B &= hf\gamma\\ \\ P_{Ay} - P_{By} &= - P_{Ay} + P_{By} = 0\\ P_{Ay} &= P_{By}\\ P_A &= \frac{P_B}{\gamma}\\ \gamma P_A &= P_B\\ \\ P(hf)\gamma &= P(hf\gamma)\\ \\ p &\propto E\\ \\ E &= mk\\ p &= mv\gamma\\ \\ E &= 2hf\\ f_1' &= (1 + V) \gamma f\\ f_2' &= (1 - V) \gamma f\\ \\ p_3 &= p_1 + p_2\\ mv\gamma &= \frac{p}{E}E_1 + \frac{p}{E}E_2\\ \frac{2hf}{k}v\gamma &= \frac{p}{E} hf(1 + V) \gamma - \frac{p}{E} hf(1 - V) \gamma\\ 2\frac{v}{k} &= 2\frac{p}{E} V\\ \\ E &= p \frac{k}{c} \end{align}

Relativistic inelastic collision

\begin{align} U_1 &= V\\ U_2 &= -V\\ U_3 &= 0\\ \\ U_1' &= \frac{V - V}{1 - V^2} = 0\\ U_2' &= \frac{- 2V}{1 + V^2}\\ U_3' &= \frac{0 - V}{1 - 0} = -V\\ \\ E &= mk\\ K &= mk\left(\frac{1}{\sqrt{1 - V^2}} - 1\right)\\ \\ m_1 &= m\\ m_2 &= m\\ m_3 &= 2m + \frac{2K}{k}\\ \\ P &= m \gamma V\\ \\ \left(2m + \frac{2K}{k}\right) \frac{1}{\sqrt{1 - V^2}}(-V) &= m\frac{1}{\sqrt{1 - \left(\frac{-2V}{1 + V^2}\right)^2}}\left(\frac{-2V}{1 + V^2}\right)\\ \left(1 + \frac{K}{km}\right)\frac{1}{\sqrt{1 - V^2}} &= \frac{1}{\left(1 + V^2\right)\sqrt{1 - 4V^2\left(\frac{1}{1 + V^2}\right)^2}}\\ 1 - V^2 &= \left(1 + V^2\right)^2\left[1 - 4V^2\left(\frac{1}{1 + V^2}\right)^2\right]\left(\frac{1}{\sqrt{1 - V^2}}\right)^2\\ 1 - V^2 &= \left(1 + V^2\right)^2\left[1 - 4V^2\left(\frac{1}{1 + V^2}\right)^2\right]\left(\frac{1}{1 - V^2}\right)\\ \left(1 - V^2\right)^2 &= \left(1 + V^2\right)^2 - 4V^2\\ \left(1 - V^2\right)^2 &= 1 + 2V^2 + V^4 - 4V^2\\ \left(1 - V^2\right)^2 &= 1 - 2V^2 + V^4\\ \left(1 - V^2\right)^2 &= \left(1 - V^2\right)^2\\ \end{align}

Relativistic inelastic collision of two objects with same mass

\begin{align} u_2' + u_1' &= 0\\\\ \frac{u_2 - v}{1 - \frac{u_2v}{c^2}} + \frac{u_1 - v}{1 - \frac{u_1v}{c^2}}&= 0\\ (u_2 - v)\left(1 - \frac{u_1v}{c^2}\right) + (u_1 - v)\left(1 - \frac{u_2v}{c^2}\right) &= 0\\ u_2 - \frac{u_1u_2v}{c^2} - v + \frac{u_1v^2}{c^2} + u_1 - \frac{u_1u_2v}{c^2} - v + \frac{u_2v^2}{c^2} &= 0\\ u_1 + u_2 - 2\frac{u_1u_2v}{c^2} - 2v + \frac{u_1v^2}{c^2} + \frac{u_2v^2}{c^2}&= 0\\ u_1 + u_2 - 2\left(\frac{u_1u_2}{c^2} + 1\right)v + \left(\frac{u_1 + u_2}{c^2}\right)v^2 &= 0\\ c^2(u_1 + u_2) - 2(u_1u_2 + c^2)v + (u_1 + u_2)v^2 &= 0\\ c^2 - 2\frac{u_1u_2 + c^2}{u_1 + u_2}v + v^2 &= 0\\ 2 v &= 2\frac{u_1u_2 + c^2}{u_1 + u_2} - 2\sqrt{\left(\frac{u_1u_2 + c^2}{u_1 + u_2}\right)^2 - c^2}\\ v &= \frac{u_1u_2 + c^2}{u_1 + u_2} - \sqrt{\left(\frac{u_1u_2 + c^2}{u_1 + u_2}\right)^2 - c^2}\\ v &= \frac{u_1u_2 + c^2}{u_1 + u_2} - \sqrt{\frac{u_1^2 u_2^2 + 2u_1u_2c^2 + c^4}{(u_1 + u_2)^2} - \frac{c^2(u_1^2 + 2u_1u_2 + u_2^2)}{(u_1 + u_2)^2}}\\ v &= \frac{u_1u_2 + c^2}{u_1 + u_2} - \frac{\sqrt{u_1^2 u_2^2 - c^2(u_1^2 + u_2^2) + c^4}}{u_1 + u_2}\\ v &= \frac{u_1u_2 - \sqrt{(u_1^2 - c^2)(u_2^2 - c^2)} + c^2}{u_1 + u_2}\\ \end{align}

Total energy

\begin{align} E &= mk\gamma\\ p &= mv\gamma\\ \\ E &= mk\gamma\\ K &= mk \left(\gamma - 1\right)\\ M &= mk\\ P &= p\frac{k}{c}\\ \\ E &= M\gamma\\ K &= M(\gamma - 1)= E - M\\ P &= M\gamma V = EV\\ \\ E^2 &= M^2\gamma^2\\ E^2 (1 - V^2)&= M^2\\ E^2 &= M^2 + E^2V^2\\ E^2 &= M^2 + P^2\\ \end{align}

Relation between energy and mass

\begin{align} K &= Fx\\ p &= Ft\\ v &= \frac{x}{t}\\ \\ \frac{K(t)}{p} &= \frac{x(t)}{t} = v(t)\\ \frac{K}{p} &= c\\ \\ M &= mk\\ \\ c &= \frac{K}{p} = \frac{E - M}{p} = \frac{P}{p} = \frac{k}{c}\\ \\ M &= mc^2\\ \\\\ F &= \frac{dp}{dt} = \frac{dK}{dx}\\ v &= \frac{dx}{dt} = \frac{dK}{dp}\\ K &= Fx = \int v \cdot dp\\ \\ p &= mv\gamma = mcV\gamma\\ \frac{V^2}{1 - V^2} &= \left(\frac{p}{mc}\right)^2\\ V^2 &= \frac{\left(\frac{p}{mc}\right)^2}{1 + \left(\frac{p}{mc}\right)^2}\\ \\ K &= c\int{\frac{\frac{p}{mc}}{\sqrt{1 + \left(\frac{p}{mc}\right)^2}}dp}\\ q &= 1 + \left(\frac{p}{mc}\right)^2 \rightarrow \frac{m^2 c^2}{2}\cdot dq = p \cdot dp\\ K &= \frac{mc^2}{2}\int{q^{-1/2} \cdot dq}\\ K &= \left. m c^2 q^{1/2} \right|_{q_1}^{q_2} = \left. m c^2 \sqrt{1 + \left(\frac{p}{mc}\right)^2}\right|_{0}^{p}\\ K &= \sqrt{(mc^2)^2 + (pc)^2} - mc^2\\ \end{align}
