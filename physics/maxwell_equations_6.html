Ampere - Maxwell law

A charged particle

\begin{align} \vec{E} &= k_eq\frac{\vec{r}}{r^3}& \vec{B} &= k_bq\frac{\vec{v} \times \vec{r}}{r^3} \end{align} \begin{align} \vec{v} &= v\hat{i}\\ \vec{r} &= \langle x - vt, y \rangle\\ \end{align} \begin{align} \vec{v} \times \vec{r} &= \det\left[\begin{array}{cc} v & 0\\ x - vt & y\\ \end{array}\right]\hat{k}\\ &= vy\hat{k}\\ \\ \nabla \times \vec{B} &= k_bqv\cdot\det\left[\begin{array}{ccc} \hat{i} & \hat{j} & \hat{k}\\ {\Large\frac{\partial}{\partial x}} & {\Large\frac{\partial}{\partial y}} & {\Large\frac{\partial}{\partial z}}\\ 0 & 0 & \Large\frac{y}{r^3} \end{array}\right]\\ &= k_bqv\left\langle \frac{\partial}{\partial y}\left(\frac{y}{r^3}\right),-\ \frac{\partial}{\partial x}\left(\frac{y}{r^3}\right), 0 \right\rangle\\ &= k_bqv\left\langle \frac{1}{r^3} - 3\frac{y^2}{r^5}, - 3\frac{y(x-vt)}{r^5} \right\rangle\\ \\ \frac{\partial \vec{E}}{\partial t} &= k_eq\left\langle \frac{\partial \vec{E_x}}{\partial t}, \frac{\partial \vec{E_y}}{\partial t} \right\rangle\\ &= k_eqv\left\langle \frac{2}{r^3} - 3 \frac{y^2}{r^5}, -3\frac{y(x - vt)}{r^5} \right\rangle\\ \\ \nabla \times \vec{B} &= \frac{k_b}{k_e}\frac{\partial \vec{E}}{\partial t} - k_bqv \frac{1}{r^3}\hat{i}\\ \vec{J} &= -qv\frac{1}{r^3}\hat{i}\\ \end{align}

A one-dimensional wire

\begin{align} \frac{d\vec{E}}{dq} &= k_e\frac{\vec{r}}{r^3}& \frac{d\vec{B}}{dq} &= k_b\frac{\vec{v} \times \vec{r}}{r^3} \end{align} \begin{align} I &= \frac{dq}{dt}& v &= \frac{dx}{dt}& \lambda &= \frac{dq}{dx}\\ \end{align} \begin{align} I &= v\lambda\\ \end{align} \begin{align} B_z &= \lambda k_bvy \int_{-\infty}^{\infty}{\frac{1}{r^3}dx} = 2k_b\frac{I}{y}\\ \end{align} \begin{align} \nabla \times B &= \det\left[\begin{array}{ccc} \hat{i} & \hat{j} & \hat{k}\\ {\Large\frac{\partial}{\partial x}} & {\Large\frac{\partial}{\partial y}} & {\Large\frac{\partial}{\partial z}}\\ 0 & 0 & B_z \end{array}\right]\\ &= \left\langle \frac{\partial B_z}{\partial y}, -\ \frac{\partial B_z}{\partial x} \right\rangle\\ &= -2\frac{\lambda k_bv}{y^2} \hat{i} = -2k_b\frac{I}{y^2} \hat{i}\\ \\ \nabla \times B &= -2k_b\frac{I}{y^2} \hat{i} = 4\pi k_b \vec{J}\\ \vec{J} &= -\ \frac{I}{2\pi y^2} \hat{i} \end{align}

Circulation of magnetic field and current

Points in the yz-plane (perpendicular to x) \begin{align} a(a_y, a_z)\\ b(b_y, b_z)\\ \end{align} \begin{align} \vec{v} &= v\hat{i}\\ \vec{r} &= \langle x - vt, y, z \rangle\\ \\ \vec{v} \times \vec{r} &= \det\left[\begin{array}{ccc} \hat{i} & \hat{j} & \hat{k}\\ v & 0 & 0\\ x - vt & y & z\\ \end{array}\right]\\ &= v \left(-z \hat{j} + y \hat{k}\right)\\ \end{align} \begin{align} \vec{B} &= \lambda k_b (\vec{v} \times \vec{r}) \int_{-\infty}^{\infty} \frac{1}{r^3} dx = 2 \lambda k_b \frac{\vec{v} \times \vec{r}}{y^2 + z^2}\\ \int_C {\vec{B} \cdot d\vec{l}} &= \int_{a_y}^{b_y}{\vec{B}(y, a_z) \cdot d\vec{y}} + \int_{a_z}^{b_z}{\vec{B}(b_y, z) \cdot d\vec{z}} + \int_{b_y}^{a_y}{\vec{B}(y, b_z) \cdot d\vec{y}} + \int_{b_z}^{a_z}{\vec{B}(a_y, z) \cdot d\vec{z}}\\ &= \int_{a_y}^{b_y}{\left[B_y(y, a_z) - B_y(y, b_z) \right]dy} + \int_{a_z}^{b_z}{\left[B_z(b_y, z) - B_z(a_y, z) \right]dz}\\ &= 2\lambda k_b v\left(\int_{a_y}^{b_y}{\left[-\ \frac{a_z}{y^2 + a_z^2} + \frac{b_z}{y^2 + b_z^2} \right]dy} + \int_{a_z}^{b_z}{\left[\frac{b_y}{b_y^2 + z^2} - \frac{a_y}{a_y^2 + z^2} \right]dz}\right)\\ &= 2 k_b I \left(\left.\left[- \arctan\frac{y}{a_z} + \arctan\frac{y}{b_z} \right]\right|_{a_y}^{b_y} + \left.\left[\arctan\frac{z}{b_y} - \arctan\frac{z}{a_y} \right]\right|_{a_z}^{b_z}\right)\\ &= \pi k_b I \left[\frac{|b_y/b_z|}{b_y/b_z} + \frac{|a_y/a_z|}{a_y/a_z} - \frac{|b_y/a_z|}{b_y/a_z} - \frac{|a_y/b_z|}{a_y/b_z} \right]\\ &= \pi k_b I \left[\text{sgn}\ \frac{b_y}{b_z} + \text{sgn}\ \frac{a_y}{a_z} - \text{sgn}\ \frac{b_y}{a_z} - \text{sgn}\ \frac{a_y}{b_z} \right]\\ \end{align} Circulation depends on wether the wire crosses or does not cross the area \begin{align} \int_C {\vec{B} \cdot d\vec{l}} &= \left\lbrace\begin{array}{rcl} 4\pi k_b I & \text{if} & a_x, a_y < 0 < b_x, b_y & \text{crosses}\\ 0 & \text{if} & \text{sgn}\ \frac{b_y}{b_z} + \text{sgn}\ \frac{a_y}{a_z} = \text{sgn}\ \frac{b_y}{a_z} + \text{sgn}\ \frac{a_y}{b_z} & \text{does not cross}\\ -4\pi k_b I & \text{if} & a_x, a_y > 0 > b_x, b_y & \text{crosses backside}\\ \end{array}\right. \end{align}
