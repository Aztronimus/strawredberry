Lagrangian

\begin{align} E &= K + P\\ \mathcal{L} &= K - P\\ \\ K &= \frac{1}{2}(E + \mathcal{L})\\ P &= \frac{1}{2}(E - \mathcal{L})\\ \\ \mathcal{L} &= \frac{dS}{dt}\\ \\ \ddot{y} &= - g\\ \dot{y} &= - gt + k_1\\ y &= - g\frac{t^2}{2} + k_1t + k_2\\ \\ y(0) &= y_0\\ y(t_0) &= 0\\ \\ y &= - g\frac{t^2}{2} + \left(g\frac{t_0}{2} - \frac{y_0}{t_0}\right)t + y_0\\ y &= y_0\left(1 - \frac{t}{t_0}\right)\\ \\ \frac{\partial{\mathcal{L}}}{\partial{y}} &= \frac{d}{dt}\frac{\partial{\mathcal{L}}}{\partial{\dot{y}}}\\ \\\\ P &= mgy\\ K &= mg(y_0 - y)\\ \\ \mathcal{L} &= mg(y_0 - 2y)\\ S &= mg\int{(y_0 - 2y)dt} = mg\left(y_0t - 2\int{y \cdot dt}\right)\\ \\ \\ \int{y \cdot dt} &= \int{\left[g\frac{t^2}{2} - \left(\frac{y_0}{t_0} + g\frac{t_0}{2}\right)t + y_0\right]dt}\\ &= \left.\left[g\frac{t^3}{6} - \left(\frac{y_0}{t_0} + g\frac{t_0}{2}\right)\frac{t^2}{2} + y_0t\right]\right|_{t_0}\\ &= g\frac{t_0^3}{6} - \left(\frac{y_0}{t_0} + g\frac{t_0}{2}\right)\frac{t_0^2}{2} + y_0t_0\\ &= g\frac{t_0^3}{6} - \frac{y_0 t_0}{2} - g\frac{t_0^3}{4} + y_0t_0\\ &= gt_0^3\left(\frac{1}{6} - \frac{1}{4}\right) + y_0t_0\left(1 - \frac{1}{2}\right)\\ &= \frac{y_0t_0}{2} - \frac{gt_0^3}{12}\\ \\ \int{y \cdot dt} &= \int{y_0\left(1 - \frac{t}{t_0}\right)dt} = \frac{y_0t_0}{2}\\ \\ \frac{y_0t_0}{2} &> \frac{y_0t_0}{2} - \frac{gt_0^3}{12}\\ \\ \\ P &= mgy\\ K &= \frac{1}{2}m\dot{y}^2\\ \\ \mathcal{L} &= \frac{1}{2}m\dot{y}^2 - mgy\\ \\ \frac{\partial{\mathcal{L}}}{\partial{y}} &= \frac{d}{dt}\frac{\partial{\mathcal{L}}}{\partial{\dot{y}}}\\ -mg &= \frac{d}{dt}m\dot{y}\\ -mgt &= m\dot{y}\\ \dot{y} &= -gt\\ \end{align}

Euler-Lagrange equation

\begin{align} S &= \int{\mathcal{L} \cdot dt}\\ q &= q_0 + n\mu\\ \dot{q} &= \dot{q}_0 + n\dot{\mu}\\ \\ \frac{\delta S}{\delta q} &= \lim_{n\mu \rightarrow 0} \frac{S(q) - S(q_0)}{n\mu} = \lim_{n \rightarrow 0} \frac{S(q) - S(q_0)}{n} = \frac{\delta{S}}{\delta n} = 0\\ \\ \frac{\delta{S}}{\delta n} &= \frac{\delta}{\delta n} \int{\mathcal{L} \cdot dt} = \int{\frac{\delta{\mathcal{L}}}{\delta n} \cdot dt} = \int{\left[\frac{\partial{\mathcal{L}}}{\partial{\dot{q}}}\dot{\mu} + \frac{\partial{\mathcal{L}}}{\partial{q}} \mu \right]dt}\\ \\ \frac{d\mathcal{L}}{dn} &= \frac{\partial{\mathcal{L}}}{\partial{\dot{q}}}\frac{d\dot{q}}{dn} + \frac{\partial{\mathcal{L}}}{\partial{q}}\frac{dq}{dn} + \frac{\partial{\mathcal{L}}}{\partial{t}}\frac{dt}{dn} = \frac{\partial{\mathcal{L}}}{\partial{\dot{q}}}\dot{\mu} + \frac{\partial{\mathcal{L}}}{\partial{q}}\mu\\ \\ \frac{d}{dt}\left(\frac{\partial\mathcal{L}}{\partial\dot{q}}\mu\right) &= \frac{\partial\mathcal{L}}{\partial\dot{q}}\dot{\mu} + \frac{d}{dt}\left(\frac{\partial\mathcal{L}}{\partial\dot{q}}\right)\mu\\ - \frac{\partial\mathcal{L}}{\partial\dot{q}}\dot{\mu} &= \frac{d}{dt}\left(\frac{\partial\mathcal{L}}{\partial\dot{q}}\right)\mu - \frac{d}{dt}\left(\frac{\partial\mathcal{L}}{\partial\dot{q}}\mu\right)\\ \\ \int{\frac{\partial\mathcal{L}}{\partial q}\mu \cdot dt} &= - \int{\frac{\partial\mathcal{L}}{\partial\dot{q}}\dot{\mu} \cdot dt}\\ \int{\frac{\partial\mathcal{L}}{\partial q}\mu \cdot dt} &= \int{\left[\frac{d}{dt}\left(\frac{\partial\mathcal{L}}{\partial\dot{q}}\right)\mu - \frac{d}{dt}\left(\frac{\partial\mathcal{L}}{\partial\dot{q}}\mu\right)\right] dt}\\ \int_{t_0}^{t_1}{\frac{\partial\mathcal{L}}{\partial q}\mu \cdot dt} &= \int_{t_0}^{t_1}{\frac{d}{dt}\left(\frac{\partial\mathcal{L}}{\partial\dot{q}}\right)\mu \cdot dt} - Waves - mass spring system

Waves - pendulum

\begin{align} x &= r\sin{u}\\ y &= r - r\cos{u}\\ v &= r \frac{du}{dt}\\ \\ K &= \frac{1}{2} mv^2 = \frac{1}{2} mr^2 \left(\frac{du}{dt}\right)^2\\ P &= mgy = mgr(1 - \cos{u})\\ \mathcal{L} &= K - P\\ &= \frac{1}{2} mr^2 \left(\frac{du}{dt}\right)^2 - mgr(1 - \cos{u})\\ \\ \frac{\partial \mathcal{L}}{\partial u} &= \frac{d}{dt} {\left( \frac{\partial \mathcal{L}}{\partial \dot{u}} \right)}\\ \frac{\partial \mathcal{L}}{\partial u} &= - mgr\sin{u}\\ \frac{d}{dt} {\left( \frac{\partial \mathcal{L}}{\partial \dot{u}} \right)} &= \frac{d}{dt} {\left( mr^2 \frac{du}{dt} \right)} = mr^2 \frac{d^2 u}{dt^2}\\ \\ - mgr\sin{u} &= mr^2 \frac{d^2 u}{dt^2}\\ \frac{d^2 u}{dt^2} &= - \frac{g}{r}\sin{u}\\ \end{align}

Gravity
